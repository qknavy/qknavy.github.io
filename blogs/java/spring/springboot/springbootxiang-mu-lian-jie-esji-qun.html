# 3、相关配置

# 3.1、配置项

在实例化RestHighLevelClient的时候有个HttpHost对象需要实例化，而这个类提供了几个构造：

  • public HttpHost(String hostname, int port, String scheme)
  • public HttpHost(String hostname, int port)
  • public static HttpHost create(String s)
  • ...

总的来说基本上就是要提供协议（http或者https等）、主机、端口号等信息，如果es集群设置了密码的话还需要提供账号和密码，所以这里简单封装到一起了：

@Data
@ConfigurationProperties(prefix = "qknavy.config" + ElasticProperties.PREFIX)
public class ElasticProperties {
    static final String PREFIX = ".elastic";
    /**
     * url:
     * eg:
     *  192.168.1.11:9200,192.168.1.12:9200,192.168.1.13:9200
     */
    private String hosts;
    /**
     * es账号
     */
    private String username;
    /**
     * es密码
     */
    private String password;
    /**
     * 协议，默认http
     */
    private String schema = "http";
}

# 3.2、初始化RestHighLevelClient

@Slf4j
@Configuration
@EnableConfigurationProperties(ElasticProperties.class)
public class ElasticConfiguration {
    @Autowired
    private ElasticProperties elasticProperties;

    @Bean
    public RestHighLevelClient restHighLevelClient() {

        String hosts = this.elasticProperties.getHosts();
        String schema = this.elasticProperties.getSchema();
        if (!StringUtils.hasText(hosts)){
            log.error("没有配置ES的地址！！！！！！");
            return null;
        }
        final CredentialsProvider credentialsProvider = new BasicCredentialsProvider();
        credentialsProvider.setCredentials(AuthScope.ANY,new UsernamePasswordCredentials(this.elasticProperties.getUsername(), this.elasticProperties.getPassword()));

        String[] hostArr = hosts.split(",");
        List<HttpHost> httpHostList = new ArrayList<>();
        HttpHost httpHost = null;
        for (int i = 0; i < hostArr.length; i++) {
            String[] arr = hostArr[i].split(":");
            httpHost = new HttpHost(arr[0], Integer.parseInt(arr[1]), schema);
            httpHostList.add(httpHost);
        }
        HttpHost[] httpHostsArr = new HttpHost[httpHostList.size()];
        for (int i = 0; i < httpHostList.size(); i++) {
            httpHostsArr[i] = httpHostList.get(i);
        }

        RestClientBuilder builder = RestClient.builder(httpHostsArr);

        if (StringUtils.hasText(this.elasticProperties.getUsername()) && StringUtils.hasText(this.elasticProperties.getPassword())) {
            //设置密码
            builder.setHttpClientConfigCallback(new RestClientBuilder.HttpClientConfigCallback() {
                @Override
                public HttpAsyncClientBuilder customizeHttpClient(HttpAsyncClientBuilder httpAsyncClientBuilder) {
                    return httpAsyncClientBuilder.setDefaultCredentialsProvider(credentialsProvider);
                }
            });
        }
        RestHighLevelClient restHighLevelClient = new RestHighLevelClient(builder);
        return restHighLevelClient;
    }
}

# 4.2、API简单使用

public void search(String index){
    SearchRequest searchRequest = new SearchRequest();
    searchRequest.indices(index);
    SearchResponse search = null;
    try {
        search = this.highLevelClient.search(searchRequest, RequestOptions.DEFAULT);
        System.out.println(JSON.toJSONString(search.getHits()));
    } catch (IOException e) {
        e.printStackTrace();
    }
}
