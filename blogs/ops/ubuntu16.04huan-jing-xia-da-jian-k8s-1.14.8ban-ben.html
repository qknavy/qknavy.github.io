# 1、准备工作

# 1.1、系统更新

sudo apt-get update
sudo apt-get upgrade

# 2、安装docker

# 2.1、docker的安装

sudo apt-get install docker.io

# 2.2、配置docker镜像加速

sudo mkdir -p /etc/docker
sudo tee /etc/docker/daemon.json <<-'EOF'
{
  "registry-mirrors": ["https://f9b9xyar.mirror.aliyuncs.com"]
}
EOF
sudo systemctl daemon-reload
sudo systemctl restart docker

# 3、修改相关的系统配置

# 3.1、更改存储驱动

docker info |grep Cgroup
sed -i 's#ExecStart=/usr/bin/dockerd#ExecStart=/usr/bin/dockerd --exec-opt native.cgroupdriver=systemd#g' /lib/systemd/system/docker.service

# 重启生效
systemctl daemon-reload
systemctl restart docker

# 3.3、关闭selinux

# 3.4、关闭swap

swapoff -a

# 3.5、配置iptables的ACCEPT规则

iptables -F && iptables -X && iptables -F -t nat && iptables -X -t nat && iptables -P FORWARD ACCEPT

2

# 3.6、设置k8s系统环境配置

cat <<EOF >  /etc/sysctl.d/k8s.conf
net.bridge.bridge-nf-call-ip6tables = 1
net.bridge.bridge-nf-call-iptables = 1
EOF

sysctl --system

# 4、安装kubeadm、kubectl、kubelet

# 4.1、添加阿里云kube源

apt-get update && apt-get install -y apt-transport-https

curl https://mirrors.aliyun.com/kubernetes/apt/doc/apt-key.gpg | apt-key add -

cat > /etc/apt/sources.list.d/kubernetes.list << EOF
deb https://mirrors.aliyun.com/kubernetes/apt/ kubernetes-xenial main
EOF

# 4.2、安装

# 5、正式安装k8s集群

# 5.1、加载所需要的镜像

# 查看所需要的镜像及版本
kubeadm config images list

# 加载镜像（kubeadm-1.14.8.tar为开篇从网盘下载的tar文件）
docker load -i kubeadm-1.14.8.tar

# 5.2、初始化master

执行下面命令前记得修改apiserver的ip

kubeadm init --pod-network-cidr=10.244.0.0/16 --apiserver-advertise-address=192.168.0.16 --service-cidr=10.1.0.0/16  --kubernetes-version v1.14.8
1
2

注意操作成功之后的控制台输出，最好复制保存一下

# 5.3、创建认证

mkdir -p $HOME/.kube
cp -i /etc/kubernetes/admin.conf $HOME/.kube/config
chown $(id -u):$(id -g) $HOME/.kube/config

#执行完毕可查看pod
kubectl get pods --all-namespaces
